#include <stdio.h>

// ここにint min(int a, int b)関数を書く
int min(int a, int b) {
  if (a < b) {
    return a;
  } else {
    return b;
  }
}

int main() {
  int a, b, c, m;

  scanf("%d", &a);
  scanf("%d", &b);
  scanf("%d", &c);

  // aとbのうち小さいものをmに代入
  m = min(a, b);

  // 三つのうち最も小さいものをmに代入
  m = min(m, c);

  printf("%d\n", m);
  return 0;
}
