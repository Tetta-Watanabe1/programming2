#include <stdio.h>

void sudoku_print(int grid[9][9]) {
  for (int i = 0; i < 9; i++) {
    if (i % 3 == 0)
      printf("+---+---+---+\n");
    for (int j = 0; j < 9; j++) {
      if (j % 3 == 0)
        printf("|");
      if (grid[i][j] == 0) {
        printf(" ");
      } else {
        printf("%d", grid[i][j]);
      }
    }
    printf("|\n");
  }
  printf("+---+---+---+\n");
}

int main() {
  int grid[9][9] = {
      {0, 2, 3, 4, 5, 6, 7, 8, 9},
      {4, 0, 6, 7, 8, 9, 1, 0, 3},
      {7, 8, 0, 1, 2, 3, 4, 5, 6},
      {2, 3, 4, 5, 0, 7, 8, 9, 0},
      {5, 6, 7, 0, 9, 1, 2, 3, 4},
      {8, 9, 1, 2, 3, 0, 5, 6, 7},
      {3, 4, 5, 6, 7, 8, 9, 1, 0},
      {6, 7, 8, 9, 0, 2, 0, 4, 5},
      {9, 1, 0, 3, 4, 5, 6, 0, 8}};

  sudoku_print(grid);
  return 0;
}
