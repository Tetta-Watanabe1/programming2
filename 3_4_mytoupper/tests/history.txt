#include <stdio.h>

/* lower2upper関数をこの辺りに定義する */
char lower2upper(char c) {
  if ('a' <= c && c <= 'z')
    return c - ('a' - 'A');
  return c;
}

int main() {
  char c;
  while ((c = getchar()) != '\n') {

    /* ここでlower2upper関数を呼び出す．引数は変数c．戻り値はcに代入する */
    c = lower2upper(c);

    printf("%c", c);
  }
  printf("\n");
  return 0;
}
