    for (int j = 0; j < 9; j++)
      scanf("%1d", &board[i][j]);

  printf("+---+---+---+\n");
  for (int i = 0; i < 9; i++) {
    for (int j = 0; j < 9; j++) {
      if (j % 3 == 0)
        printf("|");
      if (board[i][j] == 0)
        printf(" ");
      else
        printf("%d", board[i][j]);
    }
    printf("|\n");
    if ((i + 1) % 3 == 0)
      printf("+---+---+---+\n");
  }
  return 0;
}
