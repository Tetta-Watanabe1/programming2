#include <stdio.h>

#define MAX_LEN 10

// ここにnormalize関数を書く（定義する）
#include <math.h>
void normalize(float vec[], int dim) {
  float sum = 0;
  float norm;
  for (int i = 0; i < dim; i++) {
    sum += vec[i] * vec[i];
  }
  norm = sqrt(sum);
  for (int i = 0; i < dim; i++) {
    vec[i] = vec[i] / norm;
  }
}

int main() {
  float x[MAX_LEN];
  int n;
  int i;

  scanf("%d", &n);
  for (i = 0; i < n; i++) {
    float element;
    scanf("%f", &element);
    x[i] = element;
  }

  // ここでnormalize関数を呼び出す
  normalize(x, n);
  for (i = 0; i < n; i++) {
    printf("%.4f\n", x[i]);
  }
  return 0;
}
